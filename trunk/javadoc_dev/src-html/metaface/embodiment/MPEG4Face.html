001    package metaface.embodiment;
002    import java.awt.geom.*;
003    import java.awt.*;
004    import java.awt.image.*;
005    import java.io.*;
006    import java.util.*;
007    import metaface.mpeg4.*;
008    /**
009     *  Abstract class that provides interface for classes to construct MPEG-4 faces
010     *
011     *@author     Simon Beard 
012     *@version    1.0
013     *@see        SpikyMPEG4Face
014     *@see        GirlMPEG4Face
015     */
016    public abstract class MPEG4Face {
017            /**
018             *  Facing forwards
019             */
020            protected static int FORWARDS = 0;
021            /**
022             *  Facing backwards
023             */
024            protected static int BACKWARDS = 1;
025            /**
026             *  Facing left
027             */
028            protected static int LEFT = 2;
029            /**
030             *  Facing right
031             */
032            protected static int RIGHT = 3;
033    
034            /**
035             *  Direction that the ECA is facing
036             */
037            protected int facing = FORWARDS;
038    
039    
040            /**
041             *  Creates new a MPEG4Face
042             */
043            public MPEG4Face() { }
044    
045            /**
046             *  Animates the face according to the current FAP frame and FAPU information
047             *
048             *@param  frame  The current MPEG-4 FAP frame
049             */
050            public abstract void animateFace(FAPData frame);
051    
052    
053            /**
054             *  Draws the MPEG-4 face
055             *
056             *@param  g2d     The Graphics2D object to use in drawing
057             *@param  width   The width available to the face to draw
058             *@param  height  The height available to the face to draw
059             */
060            public abstract void drawFace(Graphics2D g2d, int width, int height);
061    }
