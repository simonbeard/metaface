125             *  Constructor for the IdleBehaviour object
126             *
127             *@param  fps  The frame rate of animation
128             */
129            public IdleBehaviour(int fps) {
130                    int i;
131    
132                    //set variables
133                    framerate = fps;
134                    random_numbers = new Random();
135                    default_buffer_size = 30 * fps;
136    
137                    //Set default parameters for idle animation
138                    setEyeBlinkParameters(fps * 3, fps, Math.round((float) fps / 6f), 1, Math.round((float) fps / 4f), 1);
139                    setHeadRollParameters(fps * 2, fps, fps * 2, 5, (int) ((Math.PI / 16f) * 100000f), (int) ((Math.PI / 10f) * 100000f));
140                    setEyeMovementParameters(fps, fps / 2, fps * 2, 5, (int) ((Math.PI / 8f) * 100000f), (int) ((Math.PI / 10f) * 100000f));
141    
142                    //construct buffers
143                    buffer = new Vector();
144                    fapdatabuffer = new FAPData[125];
145                    for (i = 0; i < 125; i++) {
146                            fapdatabuffer[i] = new FAPData();
147                    }
148            }

211            /**
212             *  Generate MPEG-4 FAP eye blinks
213             */
214            private void generateEyeBlinks() {
215                    int interval;
216                    int down;
217                    int hold;
218                    int up;
219                    int i;
220                    int x;
221                    int neg_pos;
222                    boolean fit;
223                    int position;
224                    FAPData frame;
225                    int frames = 125;
226    
227                    fit = true;
228    
229                    position = 0;
230    
231                    //While more eye blinks can fit in idle animation -- calculate random eye blinks based on parameters
232                    while (fit) {
233                            neg_pos = random_numbers.nextInt(2);
234                            if (neg_pos == 0) {
235                                    neg_pos = -1;
236                            }
237                            interval = (eye_blink_interval + ((int) ((float) eye_blink_interval_margin * random_numbers.nextFloat()) * neg_pos));
238                            neg_pos = random_numbers.nextInt(2);
239                            if (neg_pos == 0) {
240                                    neg_pos = -1;
241                            }
242                            down = (eye_blink_duration_down + ((int) ((float) eye_blink_duration_margin * random_numbers.nextFloat()) * neg_pos));
243                            neg_pos = random_numbers.nextInt(2);
244                            if (neg_pos == 0) {
245                                    neg_pos = -1;
246                            }
247                            hold = (eye_blink_duration_hold + ((int) ((float) eye_blink_duration_margin * random_numbers.nextFloat()) * neg_pos));
248                            neg_pos = random_numbers.nextInt(2);
249                            if (neg_pos == 0) {
250                                    neg_pos = -1;
251                            }
252                            up = (eye_blink_duration_up + ((int) ((float) eye_blink_duration_margin * random_numbers.nextFloat()) * neg_pos));
253                            if (up == 0) {
254                                    up = 1;
255                            }
256    
257                            //Set the underlying FAP structure
258                            if (interval + down + hold + up < 125 - position) {
259                                    position = position + interval;
260                                    x = 1;
261                                    for (i = position; i < down + position; i++) {
262                                            fapdatabuffer[i].setLowLevelVariables(19, 1);
263                                            fapdatabuffer[i].setLowLevelVariables(20, 1);
264                                            fapdatabuffer[i].setLowLevelVariables(21, 1);
265                                            fapdatabuffer[i].setLowLevelVariables(22, 1);
266                                            fapdatabuffer[i].setLowLevelValues(19, x * (int) (1024f / (float) down));
267                                            fapdatabuffer[i].setLowLevelValues(20, x * (int) (1024f / (float) down));
268                                            fapdatabuffer[i].setLowLevelValues(21, x * (int) (124f / (float) down));
269                                            fapdatabuffer[i].setLowLevelValues(22, x * (int) (124f / (float) down));
270                                            x++;
271                                    }
272                                    position = position + down;
273                                    x = 1;
274                                    for (i = position; i < position + hold; i++) {
275                                            fapdatabuffer[i].setLowLevelVariables(19, 1);
276                                            fapdatabuffer[i].setLowLevelVariables(20, 1);
277                                            fapdatabuffer[i].setLowLevelVariables(21, 1);
278                                            fapdatabuffer[i].setLowLevelVariables(22, 1);
279                                            fapdatabuffer[i].setLowLevelValues(19, 1024);
280                                            fapdatabuffer[i].setLowLevelValues(20, 1024);
281                                            fapdatabuffer[i].setLowLevelValues(21, 124);
282                                            fapdatabuffer[i].setLowLevelValues(22, 124);
283                                            x++;
284                                    }
285                                    position = position + hold;
286                                    x = 1;
287                                    for (i = position; i < position + up; i++) {
288                                            fapdatabuffer[i].setLowLevelVariables(19, 1);
289                                            fapdatabuffer[i].setLowLevelVariables(20, 1);
290                                            fapdatabuffer[i].setLowLevelVariables(21, 1);
291                                            fapdatabuffer[i].setLowLevelVariables(22, 1);
292                                            fapdatabuffer[i].setLowLevelValues(19, 1024 - (x * (int) (1024f / (float) up)));
293                                            fapdatabuffer[i].setLowLevelValues(20, 1024 - (x * (int) (1024f / (float) up)));
294                                            fapdatabuffer[i].setLowLevelValues(21, 124 - (x * (int) (124f / (float) up)));
295                                            fapdatabuffer[i].setLowLevelValues(22, 124 - (x * (int) (124f / (float) up)));
296                                            x++;
297                                    }
298                                    position = position + up;
299                            } else {
300                                    fit = false;
301                            }
302                    }
303            }

306            /**
307             *  Generate MPEG-4 FAP head rolls
308             */
309            private void generateHeadRoll() {
310                    int interval;
311                    int left_right;
312                    int neg_pos;
313                    int amplitude;
314                    int duration;
315                    boolean fit;
316                    int position;
317                    FAPData frame;
318                    int i;
319                    int x;
320                    int frames = 125;
321    
322                    fit = true;
323    
324                    position = 0;
325                    //While more head rolls can fit in idle animation -- calculate random head rolls based on parameters
326                    while (fit) {
327                            left_right = random_numbers.nextInt(2);
328                            if (left_right == 0) {
329                                    left_right = -1;
330                            }
331                            neg_pos = random_numbers.nextInt(2);
332                            if (neg_pos == 0) {
333                                    neg_pos = -1;
334                            }
335                            interval = (head_roll_interval + ((int) ((float) (head_roll_interval_margin * neg_pos) * random_numbers.nextFloat())));
336                            neg_pos = random_numbers.nextInt(2);
337                            if (neg_pos == 0) {
338                                    neg_pos = -1;
339                            }
340                            amplitude = left_right * (head_roll_amplitude + ((int) ((float) (head_roll_amplitude_margin * neg_pos) * random_numbers.nextFloat())));
341                            neg_pos = random_numbers.nextInt(2);
342                            if (neg_pos == 0) {
343                                    neg_pos = -1;
344                            }
345    
346                            //Set the underlying FAP structure
347                            duration = head_roll_duration + ((int) ((float) (head_roll_duration_margin * neg_pos) * random_numbers.nextFloat()));
348                            if (interval + duration + duration < 125 - position) {
349                                    position = position + interval;
350                                    x = 1;
351                                    for (i = position; i < position + duration; i++) {
352                                            fapdatabuffer[i].setLowLevelVariables(50, 1);
353                                            fapdatabuffer[i].setLowLevelValues(50, x * (int) ((float) amplitude / (float) duration));
354                                            x++;
355                                    }
356                                    position = position + duration;
357                                    x = 1;
358                                    for (i = position; i < position + duration; i++) {
359                                            fapdatabuffer[i].setLowLevelVariables(50, 1);
360                                            fapdatabuffer[i].setLowLevelValues(50, amplitude - (x * (int) ((float) amplitude / (float) duration)));
361                                            x++;
362                                    }
363                                    position = position + duration;
364                            } else {
365                                    fit = false;
366                            }
367                    }
368            }

459            /**
460             *  Generates MPEG-4 FAPs for idle behaviour
461             *
462             *@param  frames  The number of FAP frames to generate
463             *@return         A list of FAPData MPEG-4 FAPs
464             *@see            metaface.mpeg4.FAPData
465             */
466            public Vector generateIdleFAPs(int frames) {
467                    int i;
468                    Vector output;
469    
470                    output = new Vector();
471    
472                    for (i = 0; i < frames; i++) {
473                            fapdatabuffer[i].reset();
474                    }
475    
476                    generateEyeBlinks();
477                    generateHeadRoll();
478                    generateEyeMovement();
479    
480                    for (i = 0; i < frames; i++) {
481                            output.add(fapdatabuffer[i]);
482                    }
483    
484                    return (output);
485            }

488            /**
489             *  Generates MPEG-4 FAPs for transitioning to the neutral position
490             *
491             *@param  lastframe  The MPEG-4 FAP frame to transition from
492             *@param  frames     The number of frames to use in the transition
493             *@return            A list of FAPData MPEG-4 FAPs
494             *@see               metaface.mpeg4.FAPData
495             */
496            public Vector toNeutralFAPs(FAPData lastframe, int frames) {
497                    int i;
498                    int j;
499                    int val;
500                    int blend;
501                    int in1;
502                    int in2;
503                    Vector output;
504    
505                    output = new Vector();
506    
507                    //Linear transtion for all FAP values to zero over "i" frames
508                    for (i = 0; i < frames; i++) {
509                            output.add(new FAPData());
510                            ((FAPData) output.elementAt(i)).reset();
511                            if (lastframe.visemeMaskSet()) {
512                                    blend = lastframe.getVisemeBlend();
513                                    blend = blend - (int) ((float) (i + 1) * ((float) blend / (float) frames));
514                                    if (blend < 0) {
515                                            blend = 0;
516                                    }
517                                    ((FAPData) output.elementAt(i)).setVisemeValues(lastframe.getViseme1(), 0, blend);
518                                    ((FAPData) output.elementAt(i)).setVisemeVariables(1);
519                            }
520                            if (lastframe.expressionMaskSet()) {
521                                    in1 = lastframe.getExpressionIntensity1();
522                                    in2 = lastframe.getExpressionIntensity2();
523                                    in1 = in1 - (int) ((float) (i + 1) * ((float) in1 / (float) frames));
524                                    if (in1 < 0) {
525                                            in1 = 0;
526                                    }
527                                    in2 = in2 - (int) ((float) (i + 1) * ((float) in2 / (float) frames));
528                                    if (in2 < 0) {
529                                            in2 = 0;
530                                    }
531                                    ((FAPData) output.elementAt(i)).setExpressionValues(lastframe.getExpression1(), in1, lastframe.getExpression2(), in2);
532                                    ((FAPData) output.elementAt(i)).setExpressionVariables(1, 0, 0);
533                            }
534                            for (j = 3; j <= 68; j++) {
535                                    if (lastframe.lowLevelMaskSet(j)) {
536                                            val = lastframe.getLowLevelValue(j);
537                                            val = val - (int) ((float) (i + 1) * ((float) val / (float) frames));
538                                            if (val < 0) {
539                                                    val = 0;
540                                            }
541                                            ((FAPData) output.elementAt(i)).setLowLevelValues(j, val);
542                                            ((FAPData) output.elementAt(i)).setLowLevelVariables(j, 1);
543                                    }
544                            }
545                    }
546                    return (output);
547            }
