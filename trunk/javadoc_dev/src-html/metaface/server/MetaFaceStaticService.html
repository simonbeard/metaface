001    package metaface.server;
002    import java.io.*;
003    import java.net.*;
004    import java.awt.*;
005    import java.awt.event.*;
006    import java.util.*;
007    import metaface.interaction.*;
008    import metaface.dsp.*;
009    import metaface.client.*;
010    
011    /**
012     *  This class provides a static client service based on the prerendering of
013     *  VHML. Prerendered FAPs and phonemes are retrieved when needed in response to
014     *  a client's stimulus.
015     *
016     *@author     Simon Beard 
017     *@version    1.0
018     *@see        metaface.client.HTTPClient
019     *@see        metaface.client.MetaFaceHTTPClient
020     *@see        MetaFaceHTTPServer
021     *@see        VHMLOutputReader
022     */
023    public class MetaFaceStaticService extends MetaFaceService {
024    
025            /**
026             *  Constructor for the MetaFaceService object
027             *
028             *@param  interactionmanager      The interaction manager to handle user
029             *      (client machine) stimulus
030             *@param  s                       The client socket
031             *@param  httpserver              The server
032             *@param  digitalsignalprocessor  The DSP
033             *@exception  IOException         Input/Output Exception
034             */
035            public MetaFaceStaticService(InteractionManager interactionmanager, Socket s, MetaFaceHTTPServer httpserver, DigitalSignalProcessor digitalsignalprocessor) throws IOException {
036                    super(interactionmanager, s, httpserver, digitalsignalprocessor);
037            }
038    
039    
040            /**
041             *  Process a response from the interaction manager (in accordance with client
042             *  stimulus).
043             *
044             *@param  response  The response string (a VHML file)
045             */
046            protected void processResponse(String response) {
047                    VHMLOutputReader vhmloutputreader;
048    
049                    vhmloutputreader = new VHMLOutputReader(response);
050                    setFrames(vhmloutputreader.getFrames());
051                    setPhonemes(vhmloutputreader.getPhonemes());
052                    setBrowserUpdates(vhmloutputreader.getBrowserUpdates());
053                    setText(vhmloutputreader.getText());
054            }
055    }
