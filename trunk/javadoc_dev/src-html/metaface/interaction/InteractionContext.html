001    package metaface.interaction;
002    
003    import java.util.*;
004    import java.io.*;
005    /**
006     *  A utility class for keeping the current interaction context
007     *
008     *@author     Simon Beard
009     *@version    1.0
010     *@see        InteractionManager
011     */
012    public class InteractionContext {
013            /**
014             *  The states that are linked via "nextstates" to the current state
015             */
016            public static Vector linked = new Vector();
017            /**
018             *  Previous paths in which a state response was found
019             */
020            public static Stack previousstatepaths = new Stack();
021            /**
022             * The current processing type (e.g. InteractionManager.ENTRY)
023             */
024            public static int processingtype = 0;
025            /**
026             * The current stimulus type (e.g. InteractionManager.HYPERTEXT)
027             */
028            public static int stimulustype = 0;
029            /**
030             * The current stimulus string
031             */
032            public static String stimulus = "";
033    
034    
035            /**
036             *  Constructor for the InteractionContext object
037             */
038            public InteractionContext() { }
039    }
