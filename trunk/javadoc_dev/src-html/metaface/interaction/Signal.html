001    package metaface.interaction;
002    
003    import java.util.*;
004    import java.io.*;
005    /**
006     *  This class provides a signal to application code once a state is given as a
007     *  response
008     *
009     *@author     Simon Beard 
010     *@version    1.0
011     *@see        InteractionManager
012     */
013    public class Signal {
014            /**
015             *  The name attribute of the Signal element
016             */
017            private String name;
018            /**
019             *  The data type attribute of the Signal element
020             */
021            private String datatype;
022            /**
023             *  The data attribute of the Signal element
024             */
025            private String data;
026    
027    
028            /**
029             *  Constructor for the Signal object
030             *
031             *@param  namestring      The DMTL name component
032             *@param  datatypestring  The DMTL data type component
033             *@param  datastring      The DMTL data component
034             */
035            public Signal(String namestring, String datatypestring, String datastring) {
036                    name = namestring;
037                    datatype = datatypestring;
038                    data = datastring;
039            }
040    
041    
042            /**
043             *  Gets the name attribute of the Signal element
044             *
045             *@return    The name
046             */
047            protected String getName() {
048                    return (name);
049            }
050    
051    
052            /**
053             *  Gets the data type attribute of the Signal element
054             *
055             *@return    The data type
056             */
057            protected String getDataType() {
058                    return (datatype);
059            }
060    
061    
062            /**
063             *  Gets the data attribute of the Signal element
064             *
065             *@return    The data
066             */
067            protected String getData() {
068                    return (data);
069            }
070    
071    
072            /**
073             *  Sets the name attribute of the Signal element
074             *
075             *@param  namestring  The new name
076             */
077            protected void setName(String namestring) {
078                    name = namestring;
079            }
080    
081    
082            /**
083             *  Sets the dataType attribute of the Signal element
084             *
085             *@param  datatypestring  The new data type
086             */
087            protected void setDataType(String datatypestring) {
088                    data = datatypestring;
089            }
090    
091    
092            /**
093             *  Sets the data attribute of the Signal element
094             *
095             *@param  datastring  The new data
096             */
097            protected void setData(String datastring) {
098                    data = datastring;
099            }
100    
101    
102            /**
103             *  This method is called when a state is given as a response and this signal
104             *  element is encountered
105             */
106            public void execute() { }
107    
108    
109            /**
110             *  Any buiding instructions that must be completed after the constructor is
111             *  called
112             */
113            public void build() { }
114    }
