001    package metaface.dsp;
002    import java.util.*;
003    import java.io.*;
004    import java.applet.*;
005    /**
006     *  This class defines the structure by which all inheriting Digital Signal
007     *  Processors (DSPs) must adhere. A DSP takes textual phoneme information and
008     *  synthesises speech audio
009     *
010     *@author     Simon Beard 
011     *@version    1.0
012     */
013    public abstract class DigitalSignalProcessor {
014    
015            /**
016             *  Constructor for the DSP
017             */
018            public DigitalSignalProcessor() { }
019    
020    
021            /**
022             *  Sythesise phoneme information into a speech audio file
023             *
024             *@param  input  A string of phoneme information
025             *@return        A speech audio file name string
026             */
027            public abstract String synthesise(String input);
028    
029    
030            /**
031             *  <p>
032             *
033             *  Get the visual equivalent of a phoneme</p> 
034             * <IMG SRC="../../resources/visemes.jpg"/>
035             *
036             *@param  ph  The phoneme to look up
037             *@return     An MPEG-4 numbered viseme representing the phoneme
038             */
039            public abstract int getViseme(String ph);
040    
041    
042            /**
043             *  <p>
044             *
045             *  Get the visual equivalents of phonemes</p>
046             *
047             *@param  phonemeinfo  The phoneme information (phonemes and duration)
048             *@return              The MPEG-4 numbered visemes representing the phoneme
049             *      information and durations (milliseconds):<br>
050             *      [viseme1][duration1][viseme2][duration2][viseme3] ...
051             */
052            public abstract Vector phonemesToVisemes(String phonemeinfo);
053    
054    
055            /**
056             *  Get a phoneme string representing a silent duration
057             *
058             *@param  milliseconds  The duration of the silence
059             *@return               A phoneme string representing the silence
060             */
061            public abstract String silencePhonemeString(int milliseconds);
062    }
