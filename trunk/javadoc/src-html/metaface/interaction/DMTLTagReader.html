009    /**
010     *  This class contains the callback routines neccessary to read DMTL into
011     *  memory (KnowledgeBase). This format is then used by the InteractionManager.
012     *  DMTL is defined by the following DMTL structure: 
013     *  <IMG SRC="../../resources/DMTL_h.jpg"/><BR/>
014     *  <IMG SRC="../../resources/DMTL_a.jpg"/><BR/>
015     *  See the DMTL <A HREF="../../resources/DMTL.dtd">DTD</a> or and 
016     *  <A HREF="../../resources/dmtl_example.xml">example file</a>.
017     *
018     *@author     Simon Beard 
019     *@version    1.0
020     *@see        InteractionManager
021     *@see        DMTLParser
022     *@see        KnowledgeBase
023     */
024    public class DMTLTagReader extends DefaultHandler {
025            /**
026             *  Store the current topic
027             */
028            private Topic currenttopic;
029            /**
030             *  Store the default topic
031             */
032            private DefaultTopic currentdefaulttopic;
033            /**
034             *  Store the current subtopic
035             */
036            private SubTopic currentsubtopic = null;
037            /**
038             *  Store the current state
039             */
040            private State currentstate;
041            /**
042             *  Store the current stimulus
043             */
044            private Stimulus currentstimulus;
045            /**
046             *  Store the stimulus macros
047             */
048            private Macros macros;
049            /**
050             *  The knowledgebase constructed from DMTL
051             */
052            private KnowledgeBase knowledgebase;
053            /**
054             *  The current processing state of the DMTL document
055             */
056            DMTLMemoryState stateInfo;
057            /**
058             *  Temporary storage string
059             */
060            String tempchar = new String("");
061            /**
062             *  The subtopics of the current subtopic
063             */
064            Stack subtopicstack;
065    
066    
067            /**
068             *  Constructor for the DMTLTagReader object
069             *
070             *@param  filename  The file to construct a knowledge base from
071             */
072            public DMTLTagReader(String filename) {
073                    super();
074                    knowledgebase = new KnowledgeBase(filename);
075                    if (DMTLDebugger.debug) {
076                            System.out.println("constructing...");
077                    }
078                    stateInfo = new DMTLMemoryState();
079                    macros = new Macros();
080                    subtopicstack = new Stack();
081            }

084            /**
085             *  Gets the knowledge base of the DMTLTagReader object
086             *
087             *@return    The knowledge base
088             */
089            public KnowledgeBase getKnowledgeBase() {
090                    return (knowledgebase);
091            }

119             *  The parser calls this when encountering character data
120             *
121             *@param  ch                The characers encountered between tags
122             *@param  start             The starting point in the character array
123             *@param  length            The length of character data occupying the array
124             *@exception  SAXException  XML parsing Exception
125             */
126            public void characters(char[] ch, int start, int length)
127                             throws SAXException {
128                    if (DMTLDebugger.debug) {
129                            System.out.println("reading character data...");
130                    }
131                    tempchar = tempchar.concat(new String(ch, start, length));
132            }

135            /**
136             *  Parser calls this for each element in a document
137             *
138             *@param  namespaceURI      Used to resolve the namespace of the element
139             *@param  localName         The local name of the element
140             *@param  rawName           The resolved name of the element
141             *@param  atts              The attributes of the element
142             *@exception  SAXException  XML parsing Exception
143             */
144            public void startElement(String namespaceURI, String localName,
145                            String rawName, Attributes atts)
146                             throws SAXException {
147                    String key = rawName;
148                    String name;
149                    String datatype;
150                    String data;
151                    String type;
152                    String weight;
153                    String ref;
154                    String keywords;
155                    String evaluate;
156                    String statereference;
157                    String variable;
158                    String operator;
159                    String value;
160    
161                    if (DMTLDebugger.debug) {
162                            System.out.print("starting element... ");
163                    }
164    
165                    //Work out which element has been encountered
166                    if (key.compareTo("dialogue") == 0) {
167                            if (DMTLDebugger.debug) {
168                                    System.out.println("dialogue...");
169                            }
170    
171                            //Set the current processing state
172                            stateInfo.setState(DMTLMemoryState.DIALOGUE);
173                    } else if (key.compareTo("macros") == 0) {
174                            if (DMTLDebugger.debug) {
175                                    System.out.println("  macros...");
176                            }
177    
178                            //Set the current processing state
179                            stateInfo.setState(DMTLMemoryState.MACROS);
180                    } else if (key.compareTo("macro") == 0) {
181                            if (DMTLDebugger.debug) {
182                                    System.out.print("    macro... ");
183                            }
184    
185                            //Set the current processing state
186                            stateInfo.setState(DMTLMemoryState.MACRO);
187    
188                            //Get element attributes
189                            name = atts.getValue(0);
190    
191                            //Set the current processing state
192                            stateInfo.setMacroName(name);
193                            if (DMTLDebugger.debug) {
194                                    System.out.println(name);
195                            }
196                    } else if (key.compareTo("defaulttopic") == 0) {
197                            if (DMTLDebugger.debug) {
198                                    System.out.println("  defaulttopic...");
199                            }
200    
201                            //Set the current processing state
202                            stateInfo.setState(DMTLMemoryState.DEFAULTTOPIC);
203                            stateInfo.setIfDefaultTopic(true);
204    
205                            //Construct element
206                            currentdefaulttopic = new DefaultTopic();
207                    } else if (key.compareTo("topic") == 0) {
208                            if (DMTLDebugger.debug) {
209                                    System.out.print("  topic... ");
210                            }
211                            //Set the current processing state
212                            stateInfo.setState(DMTLMemoryState.TOPIC);
213    
214                            //Get attributes
215                            name = atts.getValue(0);
216    
217                            //Construct the element
218                            currenttopic = new Topic(name);
219                            if (DMTLDebugger.debug) {
220                                    System.out.println(name);
221                            }
222                    } else if (key.compareTo("subtopic") == 0) {
223                            if (DMTLDebugger.debug) {
224                                    System.out.print("    subtopic... ");
225                            }
226                            //Set the current processing state
227                            stateInfo.setState(DMTLMemoryState.SUBTOPIC);
228    
229                            //Get attributes
230                            name = atts.getValue(0);
231                            keywords = atts.getValue("keywords");
232                            evaluate = atts.getValue("evaluate");
233    
234                            //Construct the element
235                            currentsubtopic = new SubTopic(name, keywords, evaluate);
236    
237                            //Keep trach of the previos subtopics
238                            subtopicstack.push(currentsubtopic);
239                            if (DMTLDebugger.debug) {
240                                    System.out.println(name);
241                            }
242                    } else if (key.compareTo("state") == 0) {
243                            if (DMTLDebugger.debug) {
244                                    System.out.print("      state... ");
245                            }
246    
247                            //Set the current processing state
248                            stateInfo.setState(DMTLMemoryState.STATE);
249    
250                            //Get attributes
251                            name = atts.getValue("name");
252                            type = atts.getValue("type");
253    
254                            //Work out the type of State encountered and construct
255                            if (type.indexOf("linked") == 0) {
256                                    currentstate = new LinkedState(name);
257                            } else if (type.indexOf("active") == 0) {
258                                    currentstate = new ActiveState(name);
259                            } else {
260                                    currentstate = new EntryState(name);
261                            }
262                            if (DMTLDebugger.debug) {
263                                    System.out.println(name);
264                            }
265                    } else if (key.compareTo("stimulus") == 0) {
266                            if (DMTLDebugger.debug) {
267                                    System.out.println("        stimulus...");
268                            }
269                            tempchar = "";
270    
271                            //If the stimulus is inside a macro
272                            if (stateInfo.getState() == stateInfo.MACRO) {
273                                    stateInfo.setIfMacro(true);
274                            } else {
275    
276                                    //Else stimulus is inside a state
277                                    stateInfo.setIfMacro(false);
278                                    stateInfo.setIfFirstResponse(true);
279                                    stateInfo.setState(DMTLMemoryState.STIMULUS);
280                            }
281    
282                            //Get attributes
283                            type = atts.getValue(0);
284                            if (type == null) {
285                                    type = "text";
286                            }
287    
288                            //Set the current processing state
289                            stateInfo.setStimulusType(InteractionManager.TEXT);
290                            if (type.indexOf("hypertext") == 0) {
291                                    stateInfo.setStimulusType(InteractionManager.HYPERTEXT);
292                            }
293                    } else if (key.compareTo("response") == 0) {
294                            if (DMTLDebugger.debug) {
295                                    System.out.println("        response...");
296                            }
297                            tempchar = "";
298    
299                            //Set the current processing state
300                            stateInfo.setState(DMTLMemoryState.RESPONSE);
301    
302                            //Get attributes
303                            statereference = atts.getValue("statereference");
304                            type = atts.getValue("type");
305                            if (type == null) {
306                                    type = "text";
307                            }
308    
309                            //Set the current processing state
310                            stateInfo.setResponseType(Response.TEXT);
311                            if (type.indexOf("file") == 0) {
312                                    stateInfo.setResponseType(Response.FILE);
313                            }
314                            stateInfo.setResponseReference(statereference);
315                    } else if (key.compareTo("signal") == 0) {
316                            if (DMTLDebugger.debug) {
317                                    System.out.println("        signal...");
318                            }
319    
320                            //Set the current processing state
321                            stateInfo.setState(DMTLMemoryState.SIGNAL);
322    
323                            //Get attributes
324                            name = atts.getValue("name");
325                            datatype = atts.getValue("datatype");
326                            data = atts.getValue("data");
327    
328                            //Construct element within current state
329                            currentstate.addSignal(new Signal(name, datatype, data));
330                    } else if (key.compareTo("prestate") == 0) {
331                            if (DMTLDebugger.debug) {
332                                    System.out.println("        prestate...");
333                            }
334    
335                            //Set the current processing state
336                            stateInfo.setState(DMTLMemoryState.PRESTATE);
337    
338                            //Get attributes
339                            name = atts.getValue(0);
340    
341                            //Construct element within current state
342                            currentstate.addPreState(new PreState(name));
343                    } else if (key.compareTo("nextstate") == 0) {
344                            if (DMTLDebugger.debug) {
345                                    System.out.println("        nextstate...");
346                            }
347    
348                            //Set the current processing state
349                            stateInfo.setState(DMTLMemoryState.NEXTSTATE);
350    
351                            //Get attributes
352                            name = atts.getValue(0);
353    
354                            //Construct element within current state
355                            currentstate.addNextState(new NextState(name));
356                    } else if (key.compareTo("evaluate") == 0) {
357                            if (DMTLDebugger.debug) {
358                                    System.out.println("        evaluate...");
359                            }
360    
361                            //Set the current processing state
362                            stateInfo.setState(DMTLMemoryState.EVALUATE);
363    
364                            //Get attributes
365                            variable = atts.getValue(0);
366                            operator = atts.getValue(1);
367                            value = atts.getValue(2);
368    
369                            //Construct element within current state
370                            currentstate.addEvaluate(new Evaluate(variable, operator, value));
371                    }
372            }

395            /**
396             *  Parser calls this when encountering a closing tag
397             *
398             *@param  namespaceURI  Used to resolve the namespace of the element
399             *@param  localName     The local name of the element
400             *@param  rawName       The resolved name of the element
401             */
402            public void endElement(String namespaceURI, String localName,
403                            String rawName) {
404                    String key = rawName;
405                    WordChain tempwc;
406                    Vector tempvec;
407                    int i;
408    
409                    //Work out the name of the element encountered
410                    if (key.compareTo("dialogue") == 0) {
411    
412                            //Build the knowledge base
413                            knowledgebase.build();
414                    } else if (key.compareTo("macro") == 0) {
415                    } else if (key.compareTo("macros") == 0) {
416                            if (DMTLDebugger.debug) {
417                                    System.out.println("building macros");
418                            }
419    
420                            //Build the macros
421                            macros.build();
422                    } else if (key.compareTo("defaulttopic") == 0) {
423                            if (DMTLDebugger.debug) {
424                                    System.out.println("adding default topic to knowledge base");
425                            }
426    
427                            //Build and add to the knowledge base
428                            currentdefaulttopic.build();
429                            knowledgebase.addDefaultTopic(currentdefaulttopic);
430                            stateInfo.setIfDefaultTopic(false);
431                    } else if (key.compareTo("topic") == 0) {
432                            if (DMTLDebugger.debug) {
433                                    System.out.println("adding topic to knowledge base");
434                            }
435    
436                            //Build and add to the knowledge base
437                            currenttopic.build();
438                            knowledgebase.addTopic(currenttopic);
439                    } else if (key.compareTo("subtopic") == 0) {
440                            if (DMTLDebugger.debug) {
441                                    System.out.print("adding subtopic ");
442                            }
443                            SubTopic tmpsubtopic;
444    
445                            //Build
446                            currentsubtopic.build();
447    
448                            //If subtopic within another subtopic
449                            if (subtopicstack.size() >= 2) {
450    
451                                    //Take out last subtopic
452                                    subtopicstack.pop();
453                                    tmpsubtopic = (SubTopic) subtopicstack.pop();
454    
455                                    //Nest sub topic
456                                    tmpsubtopic.addSubTopic(currentsubtopic);
457                                    currentsubtopic = tmpsubtopic;
458    
459                                    //Put back into stack
460                                    subtopicstack.push(currentsubtopic);
461                                    if (DMTLDebugger.debug) {
462                                            System.out.println("to subtopic");
463                                    }
464                            } else {
465                                    if (DMTLDebugger.debug) {
466                                            System.out.println("to topic");
467                                    }
468    
469                                    //Is a root subtopic
470                                    for (i = 0; i < subtopicstack.size(); i++) {
471                                            subtopicstack.pop();
472                                    }
473    
474                                    //Add to current topic
475                                    currenttopic.addSubTopic(currentsubtopic);
476                            }
477                    } else if (key.compareTo("state") == 0) {
478                            if (DMTLDebugger.debug) {
479                                    System.out.println("adding state ");
480                            }
481    
482                            //Build
483                            currentstate.build();
484    
485                            //If state of a default topic
486                            if (stateInfo.getIfDefaultTopic()) {
487                                    if (DMTLDebugger.debug) {
488                                            System.out.println("  to default topic");
489                                    }
490    
491                                    //Add to default topic
492                                    currentdefaulttopic.addState(currentstate);
493                            } else {
494                                    if (DMTLDebugger.debug) {
495                                            System.out.println("  to subtopic");
496                                    }
497    
498                                    //Else add to current subtopic
499                                    currentsubtopic.addState(currentstate);
500                            }
501                    } else if (key.compareTo("stimulus") == 0) {
502                            if (DMTLDebugger.debug) {
503                                    System.out.print("adding stimulus " + tempchar + " ");
504                            }
505    
506                            //If stimulus of macro
507                            if (stateInfo.getIfMacro()) {
508                                    if (DMTLDebugger.debug) {
509                                            System.out.println("to macro");
510                                    }
511    
512                                    //Add to macro object
513                                    macros.addMacro(stateInfo.getMacroName(), tempchar.trim());
514                            } else {
515                                    if (DMTLDebugger.debug) {
516                                            System.out.println("to state");
517                                    }
518    
519                                    //Else is part of a state and is TEXT (natural language)
520                                    if (stateInfo.getStimulusType() == InteractionManager.TEXT) {
521                                            tempvec = new Vector();
522    
523                                            //Perform macro substitution on stimulus
524                                            tempvec = macros.substitute(tempchar.trim());
525    
526                                            //Add expanded stimuli to current state
527                                            for (i = 0; i < tempvec.size(); i++) {
528                                                    currentstimulus = new Stimulus((String) tempvec.elementAt(i), stateInfo.getStimulusType());
529                                                    currentstate.addStimulus(currentstimulus);
530                                            }
531                                    } else {
532                                            //Else is part of a state and is hypertext stimulus (or something different)
533                                            currentstimulus = new Stimulus(tempchar.trim(), stateInfo.getStimulusType());
534                                            currentstate.addStimulus(currentstimulus);
535                                    }
536                            }
537                    } else if (key.compareTo("response") == 0) {
538    
539                            //Add to the current state
540                            currentstate.addResponse(new Response(tempchar.trim(), stateInfo.getResponseReference(), stateInfo.getResponseType()));
541                    } else if (key.compareTo("evaluate") == 0) {
542                    } else if (key.compareTo("prestate") == 0) {
543                    } else if (key.compareTo("nextstate") == 0) {
544                    }
545            }
546    }
