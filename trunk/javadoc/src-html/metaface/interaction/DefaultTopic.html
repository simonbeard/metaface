001    package metaface.interaction;
002    
003    import java.util.*;
004    import java.io.*;
005    /**
006     *  This class store the DMTL default topic element and all of the subsequent
007     *  states. The “defaulttopic” element can contain zero or more “states” and is
008     *  used when the dialogue manager cannot find any matching responses to a
009     *  user’s stimulus. In this case a response is chosen randomly from one of the
010     *  states contained within the “defaulttopic” element. The “defaulttopic”
011     *  element is the last resort for dialogue management, and is used to generate
012     *  different “I don’t know” responses for a given DMTL document
013     *
014     *@author     Simon Beard 
015     *@version    1.0
016     *@see        InteractionManager
017     */
018    public class DefaultTopic {
019            /**
020             *  A list of the default topic states
021             */
022            private Vector states;
023            /**
024             *  The next default response to give
025             */
026            private int nextdefaultresponse = 0;
027    
028    
029            /**
030             *  Constructor for the DefaultTopic object
031             */
032            public DefaultTopic() {
033                    states = new Vector();
034                    nextdefaultresponse = 0;
035            }
036    
037    
038            /**
039             *  Adds a state to the default topic
040             *
041             *@param  element  The state to be added
042             */
043            public void addState(State element) {
044                    states.add(element);
045            }
046    
047    
048            /**
049             *  Gets the states within the default topic
050             *
051             *@return    The DMTL states
052             */
053            public Vector getStates() {
054                    return (states);
055            }
056    
057    
058            /**
059             *  Description of the Method
060             */
061            public void build() {
062            }
063    
064    
065            /**
066             *  Description of the Method
067             */
068            public void reset() {
069                    nextdefaultresponse = 0;
070            }
071    
072    
073            /**
074             *  Gets a default response when there are no clear matches to a user's
075             *  stimulus
076             *
077             *@return    A default response chosen
078             */
079            public Response getDefaultResponse() {
080                    int i;
081                    State tmpstate;
082    
083                    //Get the next response (try not to be predicable)
084                    if (nextdefaultresponse >= states.size()) {
085                            nextdefaultresponse = 0;
086                    }
087    
088                    //Get the state
089                    tmpstate = (State) states.elementAt(nextdefaultresponse);
090    
091                    //Increment next response
092                    nextdefaultresponse++;
093    
094                    return (tmpstate.getResponse());
095            }
096    }
