001    package metaface.interaction;
002    
003    import java.util.*;
004    import java.io.*;
005    /**
006     *  This class handles all interaction within a MetaFace application.
007     *
008     *@author     Simon Beard 
009     *@version    1.0
010     *@see        KnowledgeBase
011     *@see        InteractionContext
012     *@see        BestList
013     *@see        State
014     *@see        DefaultTopic
015     *@see        Evaluate
016     *@see        NextState
017     *@see        PreState
018     *@see        Signal
019     *@see        SubTopic
020     */
021    public class InteractionManager {
022            /**
023             *  The current stimulus type is natural language
024             */
025            public final static int TEXT = 0;
026            /**
027             *  The current stimulus type is hypertext
028             */
029            public final static int HYPERTEXT = 1;
030    
031            /**
032             *  The current processing type is all entry states
033             */
034            public final static int ALLENTRY = 0;
035            /**
036             *  The current processing type is entry within the current subtopic
037             */
038            public final static int ENTRY = 1;
039            /**
040             *  The current processing type is all linked states
041             */
042            public final static int LINKED = 2;
043            /**
044             *  The current processing type is the default topic states
045             */
046            public final static int DEFAULT = 3;
047            /**
048             *  The DMTL knowledgebases used in parsing
049             */
050            private Vector knowledgebases;
051            /**
052             *  The current interaction context
053             */
054            private InteractionContext context;
055            /**
056             *  The list of best matching states to a user's stimulus
057             */
058            private BestList bestlist;
059            /**
060             *  The path to response files
061             */
062            private String basepath;
063    
064    
065            /**
066             *  Constructor for the InteractionManager object
067             *
068             *@param  kbs  A list of knowledge bases
069             */
070            public InteractionManager(Vector kbs) {
071                    knowledgebases = kbs;
072                    context = new InteractionContext();
073                    context.processingtype = ALLENTRY;
074                    bestlist = new BestList();
075                    basepath = "";
076            }
077    
078    
079            /**
080             *  Constructor for the InteractionManager object
081             */
082            public InteractionManager() {
083                    knowledgebases = new Vector();
084                    context = new InteractionContext();
085                    context.processingtype = ALLENTRY;
086                    bestlist = new BestList();
087                    basepath = "";
088            }
089    
090    
091            /**
092             *  Sets the base path for response files
093             *
094             *@param  path  The new base path value
095             */
096            public void setBasePath(String path) {
097                    if ((path.charAt(path.length() - 1) != '\\') && (path.charAt(path.length() - 1) != '/')) {
098                            basepath = path + "/";
099                    } else {
100                            basepath = path;
101                    }
102            }
103    
104    
105            /**
106             *  Gets the base path for response files
107             *
108             *@return    The base path value
109             */
110            public String getBasePath() {
111                    return (basepath);
112            }
113    
114    
115            /**
116             *  Adds a knowledge base to the InteractionManager object
117             *
118             *@param  kb  The knowledgebase to be added
119             */
120            public void addKnowledgeBase(KnowledgeBase kb) {
121                    knowledgebases.add(kb);
122            }
123    
124    
125            /**
126             *  Gets the knowledge bases of the InteractionManager object
127             *
128             *@return    A list of knowledge bases
129             */
130            public Vector getKnowledgeBases() {
131                    return (knowledgebases);
132            }
133    
134    
135            /**
136             *  Process a user's stimulus to find a response. The follwing searching
137             *  strategy is used: <menu>
138             *  <li> 1. Search the current subtopic (the subtopic from which the last
139             *  response came).</li>
140             *  <li> 2. If no relevant responses are found then search the subtopics and
141             *  topics at the same hierarchical level. If no valid response is found, then
142             *  the next highest level is searched, and so on.</li>
143             *  <li> 3. If no relevant responses are found after searching the whole
144             *  document then a response is retrieved from “defaulttopic”.</li> </menu>
145             *
146             *@param  stimulus  The user's stimulus (e.g. hypertext or natural language)
147             *@return           The response as text, or the absolute path of a file, or
148             *      null if none is found
149             */
150            public String processUserStimulus(String stimulus) {
151                    Response response = null;
152                    Response responsetemp = null;
153                    FileReader fr;
154                    char[] buffer;
155                    String responsetext;
156                    int i;
157                    int j;
158                    int k;
159                    Stack path;
160                    Stack path2;
161                    String temp;
162                    Vector temppath;
163                    Vector temppath2;
164    
165                    //Set context
166                    context.stimulus = stimulus;
167    
168                    //Check the type of user stimulus and set context
169                    if ((stimulus.indexOf("http://") == 0) || (stimulus.indexOf("file://") == 0)) {
170                            context.stimulustype = HYPERTEXT;
171                    } else {
172                            context.stimulustype = TEXT;
173                    }
174    
175                    if (DMTLDebugger.debug) {
176                            if (context.processingtype == LINKED) {
177                                    System.out.println("\n\nProccessing LINKED states");
178                            } else {
179                                    System.out.println("\n\nProccessing ALL ENTRY states");
180                            }
181                    }
182    
183                    //For each knowledge base -- perform matches
184                    for (i = 0; i < knowledgebases.size(); i++) {
185                            bestlist = ((KnowledgeBase) knowledgebases.elementAt(i)).matchStimulus(context);
186                    }
187    
188                    //Proceed with matching process until a match is found or a default response has to be retrieved
189                    while ((bestlist.size() == 0) && (context.processingtype != DEFAULT)) {
190                            if (context.processingtype == ENTRY) {
191                                    if (DMTLDebugger.debug) {
192                                            System.out.println("\n\nProccessing ALL ENTRY states");
193                                    }
194                                    context.processingtype = ALLENTRY;
195                                    for (i = 0; i < knowledgebases.size(); i++) {
196                                            bestlist = ((KnowledgeBase) knowledgebases.elementAt(i)).matchStimulus(context);
197                                    }
198                            } else if (context.processingtype == ALLENTRY) {
199                                    if (DMTLDebugger.debug) {
200                                            System.out.println("\n\nProccessing DEFAULT states");
201                                    }
202                                    context.processingtype = DEFAULT;
203                            } else if (context.processingtype == LINKED) {
204                                    if (DMTLDebugger.debug) {
205                                            System.out.println("\n\nProccessing previous ENTRY states");
206                                    }
207                                    context.processingtype = ENTRY;
208                                    for (i = 0; i < knowledgebases.size(); i++) {
209                                            bestlist = ((KnowledgeBase) knowledgebases.elementAt(i)).matchStimulus(context);
210                                    }
211                            }
212                    }
213    
214                    response = null;
215    
216                    //If stimulus is not hypertext then get a default response
217                    if ((context.processingtype == DEFAULT) && (context.stimulustype != HYPERTEXT)) {
218                            if (DMTLDebugger.debug) {
219                                    System.out.println("Getting default response");
220                            }
221    
222                            //Get default response
223                            for (i = 0; i < knowledgebases.size(); i++) {
224                                    response = ((KnowledgeBase) knowledgebases.elementAt(i)).getDefaultResponse();
225                            }
226                            context.processingtype = ALLENTRY;
227                            context.linked.clear();
228                    } else if (bestlist.size() > 0) {
229                            //If there is a valid response
230    
231                            if (DMTLDebugger.debug) {
232                                    System.out.println("\n\n");
233                                    bestlist.print();
234                            }
235    
236                            //Reverse the path order to retrieve response
237                            if (DMTLDebugger.debug) {
238                                    System.out.println("\n\nRetrieving response");
239                            }
240                            path = bestlist.getReferenceAt(0);
241                            temppath = new Vector();
242                            j = path.size();
243                            for (i = 0; i < j; i++) {
244                                    temppath.add(path.pop());
245                            }
246                            temp = "";
247                            for (i = 0; i < j; i++) {
248                                    path.push((String) temppath.elementAt(i));
249                            }
250                            for (i = j - 1; i >= 0; i--) {
251                                    if (DMTLDebugger.debug) {
252                                            System.out.print((String) temppath.elementAt(i) + "->");
253                                    }
254                                    if (i != j - 1) {
255                                            temp = temp.concat((String) temppath.elementAt(i) + ".");
256                                    }
257                            }
258                            if (DMTLDebugger.debug) {
259                                    System.out.println("");
260                            }
261                            temp = temp.substring(0, temp.length() - 1);
262                            context.previousstatepaths.push(temp);
263    
264                            //Get the response
265                            for (i = 0; i < knowledgebases.size(); i++) {
266                                    if ((((KnowledgeBase) knowledgebases.elementAt(i)).getFilename()).indexOf((String) path.peek()) == 0) {
267                                            temp = (String) path.pop();
268                                            if (DMTLDebugger.debug) {
269                                                    System.out.println("Traversing: " + temp);
270                                            }
271                                            response = ((KnowledgeBase) knowledgebases.elementAt(i)).getResponse(path);
272                                    }
273                            }
274    
275                            //Set context for next interaction (search linked first)
276                            context.processingtype = LINKED;
277                            context.linked = bestlist.getNextStatesAt(0);
278                    } else {
279    
280                            //Set context for next interaction (no valid response found)
281                            context.processingtype = ALLENTRY;
282                            context.linked.clear();
283                    }
284    
285                    if (response == null) {
286                            return (null);
287                    }
288    
289                    //If the response has a response reference in another state -- then it must be retieved
290                    if (response.getReference() != null) {
291                            if ((response.getReference()).size() > 0) {
292                                    if (DMTLDebugger.debug) {
293                                            System.out.println("Response has state reference: Must get new response");
294                                    }
295                                    if (DMTLDebugger.debug) {
296                                            System.out.println("Using absolute path reference");
297                                    }
298    
299                                    path = bestlist.getReferenceAt(0);
300                                    path2 = response.getReference();
301    
302                                    //reverse stack order for path
303                                    temppath = new Vector();
304                                    j = path2.size();
305                                    for (i = 0; i < j; i++) {
306                                            temppath.add(path2.pop());
307                                    }
308                                    for (i = 0; i < j; i++) {
309                                            path2.push(temppath.elementAt(i));
310                                    }
311                                    for (i = j - 1; i >= 0; i--) {
312                                            if (DMTLDebugger.debug) {
313                                                    System.out.print("->" + (String) temppath.elementAt(i));
314                                            }
315                                    }
316                                    if (DMTLDebugger.debug) {
317                                            System.out.println("");
318                                    }
319    
320                                    //Search path for a reponse
321                                    responsetemp = new Response(response.getResponse(), response.getReference(), response.getType());
322                                    for (i = 0; i < knowledgebases.size(); i++) {
323                                            response = ((KnowledgeBase) knowledgebases.elementAt(i)).getResponse(path2);
324    
325                                            //if found -- exit
326                                            if (response != null) {
327                                                    i = knowledgebases.size();
328                                            }
329                                    }
330    
331                                    //If a response was not found (maybe it was a relative path)
332                                    if (response == null) {
333                                            response = responsetemp;
334                                            if (DMTLDebugger.debug) {
335                                                    System.out.println("No response found");
336                                            }
337                                            if (DMTLDebugger.debug) {
338                                                    System.out.println("Using relative path reference");
339                                            }
340    
341                                            //Make path into absolute
342                                            path = bestlist.getReferenceAt(0);
343                                            path.pop();
344                                            temppath = new Vector();
345                                            j = path.size();
346                                            for (i = 0; i < j; i++) {
347                                                    temppath.add(path.pop());
348                                            }
349                                            for (i = j - 1; i >= 0; i--) {
350                                                    if (DMTLDebugger.debug) {
351                                                            System.out.print("->" + (String) temppath.elementAt(i));
352                                                    }
353                                            }
354    
355                                            path2 = response.getReference();
356                                            temppath2 = new Vector();
357                                            k = path2.size();
358                                            for (i = 0; i < k; i++) {
359                                                    temppath2.add(path2.pop());
360                                            }
361                                            for (i = 0; i < k; i++) {
362                                                    path2.push(temppath2.elementAt(i));
363                                            }
364                                            for (i = k - 1; i >= 0; i--) {
365                                                    if (DMTLDebugger.debug) {
366                                                            System.out.print("->" + (String) temppath2.elementAt(i));
367                                                    }
368                                            }
369                                            if (DMTLDebugger.debug) {
370                                                    System.out.println("");
371                                            }
372                                            for (i = 0; i < j; i++) {
373                                                    path2.push(temppath.elementAt(i));
374                                            }
375    
376                                            //Retrieve the response
377                                            for (i = 0; i < knowledgebases.size(); i++) {
378                                                    if ((((KnowledgeBase) knowledgebases.elementAt(i)).getFilename()).indexOf((String) path2.peek()) == 0) {
379                                                            temp = (String) path2.pop();
380                                                            if (DMTLDebugger.debug) {
381                                                                    System.out.println("Traversing: " + temp);
382                                                            }
383                                                            response = ((KnowledgeBase) knowledgebases.elementAt(i)).getResponse(path2);
384                                                    }
385                                            }
386                                    }
387                            }
388                    }
389    
390                    //If response is text -- return text
391                    if (response.getType() == Response.TEXT) {
392                            if (DMTLDebugger.debug) {
393                                    System.out.println("Retrieving response TEXT: ");
394                            }
395                            return (response.getResponse());
396                    } else if (response.getType() == Response.FILE) {
397    
398                            //If the response is a file -- return the absolute path of the file
399                            if (DMTLDebugger.debug) {
400                                    System.out.println("Retrieving response FILE: ");
401                            }
402                            responsetext = basepath + (response.getResponse());
403                            return (responsetext);
404                    }
405    
406                    return (null);
407            }
