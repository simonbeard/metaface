035            private static int FLUSH = 2;

037             *  Data for sending to the client

043            private String clientmachine;

048            /**

051            private String input;

098            /**
099             *  Constructor for the MetaFaceService object
100             *
101             *@param  interactionmanager      The interaction manager to handle user
102             *      (client machine) stimulus
103             *@param  s                       The client socket
104             *@param  httpserver              The server
105             *@param  digitalsignalprocessor  The DSP
106             *@exception  IOException         Input/Output Exception
107             */
108            public MetaFaceService(InteractionManager interactionmanager, Socket s, MetaFaceHTTPServer httpserver, DigitalSignalProcessor digitalsignalprocessor) throws IOException {
109                    _httpResponse = "HTTP/1.1 200 MetaFace \nCache-Control: no-cache\nPragma: no-cache \r\n\r\n";
110                    ack = new String("ACK").getBytes();
111                    im = interactionmanager;
112                    clientmachine = s.getInetAddress().getHostAddress();
113                    output = new DataOutputStream(new BufferedOutputStream(s.getOutputStream()));
114                    output.writeBytes(_httpResponse);
115                    output.writeInt(ack.length);
116                    output.write(ack);
117                    output.flush();
118                    hs = httpserver;
119                    dsp = digitalsignalprocessor;
120                    framerate = 25;
121            }

230             */
231            public void setInput(InputStream data) {
232                    try {
233                            String inputtemp2;
234                            input = new String("");
235                            BufferedReader inputtemp = new BufferedReader(new InputStreamReader(data));
236                            inputtemp2 = inputtemp.readLine();
237                            while (inputtemp2.length() > 0) {
238                                    input = input.concat(inputtemp2);
239                                    inputtemp2 = inputtemp.readLine();
240                            }
241                            data.close();
242                    } catch (Exception e) {}
243            }

246            /**
247             *  Main processing method for the MetaFaceService object. Executed when
248             *  incoming data arrives for this service.
249             */
250            public void run() {
251                    String response;
252                    try {
253                            String s = input;
254    
255                            //Starting sequesnce
256                            if (s.indexOf("_start_") == 0) {
257                                    if (debug) {
258                                            System.out.println("starting client " + clientmachine);
259                                    }
260                                    sendData(START, null);
261                            } else if (s.indexOf("_stop_") == 0) {
262    
263                                    //Stopping sequence
264                                    if (debug) {
265                                            System.out.println("stopping client " + clientmachine);
266                                    }
267                                    sendData(FLUSH, null);
268                            } else if (s.indexOf("_exit_") == 0) {
269    
270                                    //Exit sequence
271                                    if (debug) {
272                                            System.out.println("client exiting " + clientmachine);
273                                    }
274                                    hs.removeClient(this);
275                            } else {
276                                    //Stimulus has arrived
277    
278                                    if (debug) {
279                                            System.out.println("processing stimulus " + input + " " + clientmachine);
280                                    }
281    
282                                    //Get a response to the stimulus
283                                    response = im.processUserStimulus(s.trim());
284                                    if (debug) {
285                                            System.out.println("retrieve " + response);
286                                    }
287                                    sendData(VHML, response);
288                            }
289                    } catch (Exception e) {
290                            e.printStackTrace();
291                    }
292            }

295            /**
296             *  Sends data to the client
297             *
298             *@param  flag  The type of data to send (START, VHML, FLUSH)
299             *@param  data  The data string to send
300             */
301            private synchronized void sendData(int flag, String data) {
302                    Vector fapvec;
303                    String pi;
304                    String header;
305                    Vector visemes;
306                    int i;
307                    int j;
308                    String temp;
309                    byte[] bytes;
310                    byte[] bytes2;
311                    FAPData tempfap;
312                    char[] buf;
313                    String grouping;
314                    String fap;
315                    String ph;
316                    String html;
317                    String speechtext;
318                    String fapdata;
319                    Enumeration keys;
320                    String target;
321                    String url;
322                    VHMLOutputReader vhmloutputreader;
323                    VHMLParser vhmlparser;
324    
325                    //Start sequence
326                    if (flag == START) {
327                            try {
328    
329                                    //Send silence
330                                    pi = dsp.silencePhonemeString(1000);
331    
332                                    //Send FAPs of same length
333                                    fapvec = new Vector();
334                                    visemes = dsp.phonemesToVisemes(pi + "\n");
335                                    fapvec = FAPData.visemesToFAPString(visemes, 0f, framerate, true);
336    
337                                    //Get a FAP header
338                                    header = (String) fapvec.elementAt(0);
339                                    header = header.substring(header.indexOf("\n") + 1);
340                                    fapvec = new Vector();
341    
342                                    //Send the header
343                                    try {
344                                            bytes = header.getBytes();
345                                            output.writeByte((byte) MetaFaceHTTPClient.HEADER);
346                                            output.writeInt(bytes.length);
347                                            output.write(bytes);
348                                    } catch (IOException e) {
349                                            e.printStackTrace();
350                                    }
351    
352                                    //Send the FAPs
353                                    for (j = 0; j < framerate; j++) {
354                                            try {
355                                                    tempfap = new FAPData();
356                                                    bytes = tempfap.getBitMaskBytes();
357                                                    bytes2 = tempfap.getValueBytes(j);
358                                                    output.writeByte((byte) MetaFaceHTTPClient.FAP);
359                                                    output.writeInt(bytes.length + bytes2.length);
360                                                    output.write(bytes);
361                                                    output.write(bytes2);
362                                                    output.flush();
363                                            } catch (IOException e) {
364                                                    e.printStackTrace();
365                                            }
366                                    }
367                                    //Flush FAPs
368                                    try {
369                                            bytes = "\0\n".getBytes();
370                                            output.writeByte((byte) MetaFaceHTTPClient.FAP);
371                                            output.writeInt(bytes.length);
372                                            output.write(bytes);
373                                            output.flush();
374                                    } catch (IOException e) {
375                                            e.printStackTrace();
376                                    }
377                                    //Send phoneme information
378                                    try {
379                                            bytes = pi.getBytes();
380                                            output.writeByte((byte) MetaFaceHTTPClient.PHONEMES);
381                                            output.writeInt(bytes.length);
382                                            output.write(bytes);
383                                            bytes = "\0\n".getBytes();
384                                            output.writeByte((byte) MetaFaceHTTPClient.PHONEMES);
385                                            output.writeInt(bytes.length);
386                                            output.write(bytes);
387                                            output.flush();
388                                    } catch (IOException e) {
389                                            e.printStackTrace();
390                                    }
391                            } catch (Exception e) {
392                                    System.err.println("Server sending error");
393                                    e.printStackTrace();
394                            }
395                    } else if (flag == VHML) {
396                            //Sending VHML data
397    
398                            if (data != null) {
399                                    faps = new Vector();
400                                    phonemes = new Vector();
401                                    speechtext = "";
402                                    browserupdates = new Hashtable();
403    
404                                    //Get the FAPs, phonemes, text, URLs, and targets
405                                    processResponse(hs.getVHMLFilePath() + "/" + data);
406    
407                                    //Send URLs and targets
408                                    try {
409                                            keys = browserupdates.elements();
410                                            while (keys.hasMoreElements()) {
411                                                    target = (String) keys.nextElement();
412                                                    url = (String) browserupdates.get(target);
413                                                    bytes = (url + "->" + target).getBytes();
414                                                    output.writeByte((byte) MetaFaceHTTPClient.HTMLREF);
415                                                    output.writeInt(bytes.length);
416                                                    output.write(bytes);
417                                                    output.flush();
418                                            }
419    
420                                            //Send text
421                                            bytes = speechtext.getBytes();
422                                            output.writeByte((byte) MetaFaceHTTPClient.TEXT);
423                                            output.writeInt(bytes.length);
424                                            output.write(bytes);
425                                            output.flush();
426    
427                                            //Send FAPs
428                                            for (i = 0; i < faps.size(); i++) {
429                                                    tempfap = (FAPData) faps.elementAt(i);
430                                                    bytes = tempfap.getBitMaskBytes();
431                                                    bytes2 = tempfap.getValueBytes(i);
432                                                    output.writeByte((byte) MetaFaceHTTPClient.FAP);
433                                                    output.writeInt(bytes.length + bytes2.length);
434                                                    output.write(bytes);
435                                                    output.write(bytes2);
436                                                    output.flush();
437                                            }
438                                            bytes = "\0\n".getBytes();
439                                            output.writeByte((byte) MetaFaceHTTPClient.FAP);
440                                            output.writeInt(bytes.length);
441                                            output.write(bytes);
442                                            output.flush();
443    
444                                            //Send phoneme information
445                                            ph = "";
446                                            for (i = 0; i < phonemes.size(); i++) {
447                                                    ph = ph.concat(((String) phonemes.elementAt(i)) + "\n");
448                                            }
449                                            bytes = ph.getBytes();
450                                            output.writeByte((byte) MetaFaceHTTPClient.PHONEMES);
451                                            output.writeInt(bytes.length);
452                                            output.write(bytes);
453                                            output.flush();
454                                            bytes = "\0\n".getBytes();
455                                            output.writeByte((byte) MetaFaceHTTPClient.PHONEMES);
456                                            output.writeInt(bytes.length);
457                                            output.write(bytes);
458                                            output.flush();
459                                    } catch (IOException e) {
460                                            e.printStackTrace();
461                                    }
462                            }
463                    } else if (flag == FLUSH) {
464    
465                            //Flush sequence (send /0 for streaming FAPs and phoneme information)
466                            try {
467                                    bytes = "\0\n".getBytes();
468                                    output.writeByte((byte) MetaFaceHTTPClient.PHONEMES);
469                                    output.writeInt(bytes.length);
470                                    output.write(bytes);
471                                    output.flush();
472                                    output.writeByte((byte) MetaFaceHTTPClient.FAP);
473                                    output.writeInt(bytes.length);
474                                    output.write(bytes);
475                                    output.flush();
476                            } catch (Exception e) {
477                                    System.out.println(e + " server sending error");
478                            }
479                    }
480            }
