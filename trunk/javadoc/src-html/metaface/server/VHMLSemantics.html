390

394                            return true;
395                    }

400            /**
401             *  Gets a low level value for an MPEG-4 FAP
402             *
403             *@param  faps    The MPEG-4 FAPs to use
404             *@param  fapnum  The FAP from the list to use
405             *@param  bitord  The bit of the FAP to use
406             *@return         The low level value value corresonding to the bit
407             */
408            private int getLowLevelValue(Vector faps, int fapnum, int bitord) {
409                    int skip = 0;
410                    int x;
411                    int i;
412                    String fapvalue;
413                    String temp;
414    
415                    if (!isBitSet(faps, fapnum, bitord)) {
416                            return (0);
417                    }
418                    if (isBitSet(faps, fapnum, 1)) {
419                            skip = skip + 4;
420                    }
421                    if (isBitSet(faps, fapnum, 2)) {
422                            skip = skip + 6;
423                    }
424                    skip = skip + (bitord - 2);
425                    fapvalue = (String) faps.elementAt(fapnum + 1);
426                    x = 0;
427                    i = 0;
428                    while ((i < skip) && (x < fapvalue.length())) {
429                            if (fapvalue.charAt(x) == ' ') {
430                                    i++;
431                            }
432                            x++;
433                    }
434                    i = x;
435                    while ((fapvalue.charAt(i) != ' ') || (fapvalue.charAt(i) != ' ')) {
436                            i++;
437                    }
438                    temp = fapvalue.substring(x, i);
439                    return (new Integer(temp).intValue());
440            }

444             *  Seats a low level value for an MPEG-4 FAP
445             *
446             *@param  faps    The MPEG-4 FAPs to set
447             *@param  fapnum  The FAP from the list to set
448             *@param  bitord  The bit of the FAP to set
449             *@param  vals    The new lowLevelFAPs value
450             *@return         The modified FAPs
451             */
452            private Vector setLowLevelFAPs(Vector faps, int fapnum, int bitord, String vals) {
453                    String fapmask;
454                    String fapvalue;
455                    char[] fapmarray;
456                    int skip = 0;
457                    int i;
458                    int x;
459                    boolean remove = true;
460                    String temp;
461    
462                    fapmask = (String) faps.elementAt(fapnum);
463                    fapvalue = (String) faps.elementAt(fapnum + 1);
464                    faps.removeElementAt(fapnum);
465                    faps.removeElementAt(fapnum);
466                    fapmarray = fapmask.toCharArray();
467    
468                    if (!isBitSet(faps, fapnum, bitord)) {
469                            remove = false;
470                    }
471                    if (isBitSet(faps, fapnum, 1)) {
472                            skip = skip + 4;
473                    }
474                    if (isBitSet(faps, fapnum, 2)) {
475                            skip = skip + 6;
476                    }
477                    skip = skip + (bitord - 2);
478    
479                    fapmarray[(2 * bitord) - 2] = '1';
480                    fapmask = new String(fapmarray);
481    
482                    x = 0;
483                    i = 0;
484                    while ((i < skip) && (x < fapvalue.length())) {
485                            if (fapvalue.charAt(x) == ' ') {
486                                    i++;
487                            }
488                            x++;
489                    }
490                    i = x;
491    
492                    if (remove) {
493                            while ((fapvalue.charAt(i) != ' ') || (fapvalue.charAt(i) != ' ')) {
494                                    i++;
495                            }
496                    }
497    
498                    temp = fapvalue.substring(0, x);
499                    temp = temp.concat(" " + vals);
500                    temp = temp.concat(fapvalue.substring(i, fapvalue.length()));
501    
502                    faps.insertElementAt(fapmask, fapnum);
503                    faps.insertElementAt(temp, fapnum + 1);
504                    return (faps);
505            }

595            /**
596             *  Apply eye blinks to MPEG-4 FAPs
597             *
598             *@param  faps       The FAPs to apply blinks
599             *@param  fromframe  The index to start blinks from
600             *@param  toframe    The index to end blinks at
601             *@return            The modified FAPs
602             */
603            public Vector applyBlinking(Vector faps, int fromframe, int toframe) {
604                    int interval;
605                    int down;
606                    int hold;
607                    int up;
608                    int i;
609                    int x;
610                    int neg_pos;
611                    boolean fit;
612                    int position;
613                    FAPData frame;
614                    int frames = toframe - fromframe;
615    
616                    fit = true;
617    
618                    position = fromframe;
619                    while (fit) {
620                            neg_pos = random_numbers.nextInt(2);
621                            if (neg_pos == 0) {
622                                    neg_pos = -1;
623                            }
624                            interval = (eye_blink_interval + ((int) ((float) eye_blink_interval_margin * random_numbers.nextFloat()) * neg_pos));
625                            neg_pos = random_numbers.nextInt(2);
626                            if (neg_pos == 0) {
627                                    neg_pos = -1;
628                            }
629                            down = (eye_blink_duration_down + ((int) ((float) eye_blink_duration_margin * random_numbers.nextFloat()) * neg_pos));
630                            neg_pos = random_numbers.nextInt(2);
631                            if (neg_pos == 0) {
632                                    neg_pos = -1;
633                            }
634                            hold = (eye_blink_duration_hold + ((int) ((float) eye_blink_duration_margin * random_numbers.nextFloat()) * neg_pos));
635                            neg_pos = random_numbers.nextInt(2);
636                            if (neg_pos == 0) {
637                                    neg_pos = -1;
638                            }
639                            up = (eye_blink_duration_up + ((int) ((float) eye_blink_duration_margin * random_numbers.nextFloat()) * neg_pos));
640                            if (up == 0) {
641                                    up = 1;
642                            }
643                            if (interval + down + hold + up < faps.size() - position) {
644                                    position = position + interval;
645                                    x = 1;
646                                    for (i = position; i < down + position; i++) {
647                                            frame = (FAPData) faps.elementAt(i);
648                                            frame.setLowLevelVariables(19, 1);
649                                            frame.setLowLevelVariables(20, 1);
650                                            frame.setLowLevelVariables(21, 1);
651                                            frame.setLowLevelVariables(22, 1);
652                                            frame.setLowLevelValues(19, x * (int) (1024f / (float) down));
653                                            frame.setLowLevelValues(20, x * (int) (1024f / (float) down));
654                                            frame.setLowLevelValues(21, x * (int) (124f / (float) down));
655                                            frame.setLowLevelValues(22, x * (int) (124f / (float) down));
656                                            faps.remove(i);
657                                            faps.insertElementAt(frame, i);
658                                            x++;
659                                    }
660                                    position = position + down;
661                                    x = 1;
662                                    for (i = position; i < position + hold; i++) {
663                                            frame = (FAPData) faps.elementAt(i);
664                                            frame.setLowLevelVariables(19, 1);
665                                            frame.setLowLevelVariables(20, 1);
666                                            frame.setLowLevelVariables(21, 1);
667                                            frame.setLowLevelVariables(22, 1);
668                                            frame.setLowLevelValues(19, 1024);
669                                            frame.setLowLevelValues(20, 1024);
670                                            frame.setLowLevelValues(21, 124);
671                                            frame.setLowLevelValues(22, 124);
672                                            faps.remove(i);
673                                            faps.insertElementAt(frame, i);
674                                            x++;
675                                    }
676                                    position = position + hold;
677                                    x = 1;
678                                    for (i = position; i < position + up; i++) {
679                                            frame = (FAPData) faps.elementAt(i);
680                                            frame.setLowLevelVariables(19, 1);
681                                            frame.setLowLevelVariables(20, 1);
682                                            frame.setLowLevelVariables(21, 1);
683                                            frame.setLowLevelVariables(22, 1);
684                                            frame.setLowLevelValues(19, 1024 - (x * (int) (1024f / (float) up)));
685                                            frame.setLowLevelValues(20, 1024 - (x * (int) (1024f / (float) up)));
686                                            frame.setLowLevelValues(21, 124 - (x * (int) (124f / (float) up)));
687                                            frame.setLowLevelValues(22, 124 - (x * (int) (124f / (float) up)));
688                                            faps.remove(i);
689                                            faps.insertElementAt(frame, i);
690                                            x++;
691                                    }
692                                    position = position + up;
693                            } else {
694                                    fit = false;
695                            }
696                    }
697    
698                    return (faps);
699            }
