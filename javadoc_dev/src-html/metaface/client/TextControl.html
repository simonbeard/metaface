001    package metaface.client;
002    import javax.swing.*;
003    import javax.swing.event.*;
004    import java.awt.event.*;
005    import java.util.*;
006    import java.awt.*;
007    import java.awt.geom.*;
008    
009    /**
010     *  This class is allows text field input for an applet
011     *
012     *@author     Simon Beard 
013     *@version    1.0
014     */
015    public class TextControl extends javax.swing.JPanel {
016            /**
017             *  The text field used for dialogue input
018             */
019            private JTextField userText;
020            /**
021             *  The dimension of the object
022             */
023            private Dimension dim;
024            /**
025             *  Whether the text field is hidden
026             */
027            private boolean hidden = false;
028    
029    
030            /**
031             *  Creates new TextControl for dialogue management input
032             *
033             *@param  width  width of the TextControl object
034             */
035            public TextControl(int width) {
036                    super();
037                    this.setBackground(Color.white);
038                    userText = new JTextField();
039                    userText.setBorder(
040                                    BorderFactory.createCompoundBorder(
041                                    BorderFactory.createCompoundBorder(
042                                    BorderFactory.createTitledBorder("Type Here:"),
043                                    BorderFactory.createEmptyBorder(5, 5, 5, 5)),
044                                    userText.getBorder()));
045    
046                    userText.setBackground(new Color(255, 255, 255));
047                    dim = userText.getPreferredSize();
048                    userText.setPreferredSize(new Dimension(width, (int) (dim.getHeight())));
049                    userText.setMinimumSize(new Dimension(width, (int) (dim.getHeight())));
050                    this.setLayout(new BorderLayout(0, 0));
051                    this.add(userText, BorderLayout.CENTER);
052            }
053    
054    
055            /**
056             *  Hide the text field
057             */
058            public void hide() {
059                    userText.setVisible(false);
060            }
061    
062    
063            /**
064             *  Show the text field
065             */
066            public void show() {
067                    userText.setVisible(true);
068            }
069    
070    
071            /**
072             *  Sets the textColor attribute of the TextControl object
073             *
074             *@param  c  The new textColor value
075             */
076            public void setTextColor(Color c) {
077                    userText.setBackground(c);
078            }
079    
080    
081            /**
082             *  Gets the height attribute of the TextControl object
083             *
084             *@return    The height value
085             */
086            public int getHeight() {
087                    return ((int) dim.getHeight());
088            }
089    
090    
091            /**
092             *  Gets the textField attribute of the TextControl object
093             *
094             *@return    The textField value
095             */
096            public JTextField getTextField() {
097                    return (userText);
098            }
099    }
