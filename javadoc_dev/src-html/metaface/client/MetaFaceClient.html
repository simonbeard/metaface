183            /**
184             *  Process an MPEG-4 facial animation header (get the frame rate)
185             *
186             *@param  fapheader  The MPEG-4 header
187             */
188            protected synchronized void processMPEG4Header(String fapheader) {
189                    int ptr;
190                    String temp;
191    
192                    ptr = 0;
193    
194                    //Skip to frame rate and set class variable
195                    ptr = fapheader.indexOf(" ", ptr);
196                    ptr = fapheader.indexOf(" ", ptr + 1);
197                    temp = fapheader.substring(ptr + 1, fapheader.indexOf(" ", ptr + 1));
198                    if (new Integer(temp).intValue() == 0) {
199                            temp = String.valueOf(25);
200                    }
201                    setVideoFrameRate(new Integer(temp).intValue());
202            }

205            /**
206             *  This method updates a table of target frames and their current URL content.
207             *  This method is called by the MetaFace Applet receiving JavaScript
208             *  notifications.
209             *
210             *@param  url     The URL displayed within the target frame
211             *@param  target  The target frame
212             */
213            public synchronized void updateWebPage(String url, String target) {
214                    String oldurl;
215    
216                    //If target not in table -- add
217                    if (!webpages.containsKey("target")) {
218                            webpages.put(target, url);
219                    } else {
220                            //Else see if URL has been updated
221                            oldurl = (String) webpages.get(target);
222                            if (oldurl.indexOf(url) != 0) {
223                                    webpages.put(target, url);
224                            } else {
225                                    webpages.remove(target);
226                            }
227                    }
228            }

231            /**
232             *  The client polls for changes to the web pages table and processes
233             */
234            public synchronized void pollWebPages() {
235                    Enumeration keys;
236                    String target;
237                    String url;
238    
239                    //Check entire table
240                    keys = webpages.keys();
241                    while (keys.hasMoreElements()) {
242    
243                            //Use URLs as stimulus to interaction manager
244                            target = (String) (keys.nextElement());
245                            url = (String) (webpages.get(target));
246                            webpages.remove(target);                        
247                            processUserStimulus(url);
248                    }
249            }

335            /**
336             *  Obtain phonemes from the buffer
337             *
338             *@return    Phonemes for synthesising speech audio
339             */
340            protected synchronized Vector readPhonemesFromBuffer() {
341                    Vector output;
342                    int i;
343                    int j;
344                    output = new Vector();
345    
346                    //check if buffer empty
347                    if (phonemedatabuffer.size() == 0) {
348                            return output;
349                    } else if (phonemedatabuffer.size() > MAX_PHONEMES) {
350    
351                            //Get maximum amount of phonemes
352                            for (i = 0; i < MAX_PHONEMES; i++) {
353                                    output.add(phonemedatabuffer.remove(0));
354                            }
355                    } else {
356    
357                            //Get all phonemes
358                            j = phonemedatabuffer.size();
359                            for (i = 0; i < j; i++) {
360                                    output.add(phonemedatabuffer.remove(0));
361                            }
362                    }
363                    return (output);
364            }

631            protected synchronized void idleBufferingScheme() {
632    
633                    //If no audio being played, and no phomemes in the buffer
634                    if ((!line.isActive()) && (phonemedatabuffer.size() == 0)&& (fapdatabuffer.size() <= 5)) {
635                            
636                            resetSynchronisation();
637                            
638                            //Idle behaviour
639                            idleAnimation(125);
640

642                            addAudioSilence(5000);                  
643                            
644                            //Do clean up
645                            System.gc();
646                    }

652             *
653             *@param  filestr    A partial file string to match against potential file to
654             *      delete
655             *@param  extension  The extension of files to delete
656             */
657            protected synchronized void cleanTemporaryFiles(String filestr, String extension) {
658                    String temp = null;
659                    File tempfile;
660                    File[] list;
661                    int i;
662    
663                    //Create temporary file and get directory structure
664                    try {
665                            temp = File.createTempFile(filestr, extension).toString();
666                    } catch (IOException e) {
667                            e.printStackTrace();
668                            System.runFinalization();
669                    }
670    
671                    //Get the files in this directory
672                    temp = temp.substring(0, temp.indexOf(filestr) - 1);
673                    tempfile = new File(temp);
674                    if (tempfile.isDirectory()) {
675                            list = tempfile.listFiles();
676                            for (i = 0; i < list.length; i++) {
677    
678                                    //If match criteria, then delete
679                                    if (((list[i].toString()).indexOf(filestr) >= 0) && ((list[i].toString()).indexOf(extension) >= 0)) {
680                                            list[i].delete();
681                                    }
682                            }
683                            list = null;
684                    }
685                    tempfile = null;
686            }

747            /**
748             *  Plays ECA speech audio
749             */
750            protected synchronized void playAudio() {
751                    String filename = null;
752                    String buffer = "";
753                    Vector ph;
754                    int lines = 0;
755                    int i;
756                    int toWrite;
757                    int bytesRead = 0;
758    
759                    //If a new WAV file is needed
760                    if (newfile) {
761                            if ((audioFile != null)) {
762                                    try {
763                                            audioInputStream.close();
764                                    } catch (Exception e) {
765                                            e.printStackTrace();
766                                    }
767                                    audioFile = null;
768                            }
769    
770                            //Get phonemes and synthesise WAV file
771                            ph = readPhonemesFromBuffer();
772                            for (i = 0; i < ph.size(); i++) {
773                                    buffer = buffer.concat(((String) ph.elementAt(i)) + "\n");
774                            }
775                            if (buffer == null) {
776                                    return;
777                            }
778    
779                            filename = synthesiseWaveform(buffer);
780    
781                            if (filename == null) {
782                                    return;
783                            }
784    
785                            //Update state variables based on new WAV file
786                            newfile = false;
787                            audioFile = new File(filename);
788                            try {
789                                    audioInputStream = AudioSystem.getAudioInputStream(audioFile);
790                                    audioFrameRate = audioInputStream.getFormat().getFrameRate();
791                                    audioFrameSize = audioInputStream.getFormat().getFrameSize();
792                                    afpvf = audioFrameRate / videoFrameRate;
793    
794                                    //If first file being played -- initialise the audio line of the application
795                                    if (firstfile) {
796                                            firstfile = false;
797                                            info = new DataLine.Info(SourceDataLine.class, audioInputStream.getFormat());
798                                            line = (SourceDataLine) AudioSystem.getLine(info);
799                                            line.open();
800                                            line.start();
801                                    }
802                            } catch (UnsupportedAudioFileException e) {
803                                    System.out.println("Can't open audio input stream (Unsupported file)");
804                                    e.printStackTrace();
805                            } catch (IOException e) {
806                                    System.out.println("Can't open audio input stream (IO Exception)");
807                                    e.printStackTrace();
808                            } catch (LineUnavailableException e) {
809                                    System.out.println("Source data line not available");
810                                    e.printStackTrace();
811                            }
812                    }
813    
814                    //Read from the WAV file and place bytes in audio buffer
815                    try {
816    
817                            //Get how many bytes can be read and placed without blocking
818                            toWrite = line.available();
819                            if (toWrite > 0) {
820                                    if (toWrite > 100000) {
821                                            toWrite = 100000;
822                                    }
823    
824                                    //Get and write bytes
825                                    if ((bytesRead = audioInputStream.read(audioBytes, 0, toWrite)) != -1) {
826                                            line.write(audioBytes, 0, bytesRead);
827                                    } else {
828                                            newfile = true;
829                                            cleanFiles();
830                                    }
831                            }
832                    } catch (IOException e) {
833                            System.out.println("Problem with audio line");
834                            e.printStackTrace();
835                    }
836            }

839            /**
840             *  Determines if the current FAP frame is early based on the current frame of
841             *  the audio line buffer
842             *
843             *@param  numFrames  The number of frames rendered
844             *@return            Whether FAP is early
845             */
846            private synchronized boolean frameEarly(int numFrames) {
847                    if ((float) numFrames * afpvf > (float) line.getFramePosition() + afpvf) {
848                            return (true);
849                    }
850                    return (false);
851            }
852    
853    
854            /**
855             *  Determines if the current FAP late is early based on the current frame of
856             *  the audio line buffer
857             *
858             *@param  numFrames  The number of frames rendered
859             *@return            Whether FAP is late
860             */
861            private synchronized boolean frameLate(int numFrames) {
862                    if (((float) numFrames * afpvf < (float) line.getFramePosition())) {
863                            return (true);
864                    }
865                    return (false);
866            }

869            /**
870             *  Renders the MPEG-4 FAPs in the buffer and synchronises with speech audio
871             */
872            protected synchronized void renderAnimation() {
873                    if (!tooearly) {
874                            frame = readFrameFromBuffer();
875                            if (frame.isNull()) {
876                                    return;
877                            }
878                    }
879    
880                    // If frame is early -- return
881                    if (frameEarly(numFrames)) {
882                            tooearly = true;
883                            return;
884                    }
885    
886                    // While audio not between prev and this frame -- get next frame
887                    while (frameLate(numFrames) && (!tooearly)) {
888                            numFrames++;
889                            frame = readFrameFromBuffer();
890                            tooearly = false;
891                            if (frame.isNull()) {
892                                    return;
893                            }
894                    }
895    
896                    animateDrawingArea(frame);
897                    numFrames++;
898                    tooearly = false;
899            }
