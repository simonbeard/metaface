001    package metaface.interaction;
002    
003    import java.util.*;
004    /**
005     *  BestListItem is a structure of information for elements of a BestList vector
006     *
007     *@author     Simon Beard
008     *@version    1.0
009     *@see        BestList
010     */
011    class BestListItem {
012            /**
013             *  The score
014             */
015            private float score;
016            /**
017             *  The state reference (full path)
018             */
019            private Stack statereference;
020            /**
021             *  A list of DMTL "next states" to this one
022             */
023            private Vector nextstates;
024    
025    
026            /**
027             *  Creates a new BestListItem
028             *
029             *@param  in_score           The matching score
030             *@param  in_statereference  The refernce (full path) to the state
031             *@param  in_nextstates      Description of the Parameter
032             */
033            public BestListItem(float in_score, Stack in_statereference, Vector in_nextstates) {
034                    score = in_score;
035                    statereference = in_statereference;
036                    nextstates = in_nextstates;
037            }
038    
039    
040            /**
041             *  Gets the score of this item
042             *
043             *@return    The score
044             */
045            public float getScore() {
046                    return score;
047            }
048    
049    
050            /**
051             *  Gets the reference (full path) of the state
052             *
053             *@return    The state reference
054             */
055            public Stack getStateReference() {
056                    return (Stack) statereference.clone();
057            }
058    
059    
060            /**
061             *  Gets the DMTL "next states" of this state
062             *
063             *@return    The DMTL next states
064             */
065            public Vector getNextStates() {
066                    return (Vector) nextstates.clone();
067            }
068    }
