141            /**
142             *  Constructor for the SpeechBubble object
143             *
144             *@param  width            width of speech bubble
145             *@param  height           height of speech bubble
146             *@param  direction        direction of speech bubble peak (LEFT or DOWN)
147             *@param  text             text to display in the speech bubble
148             *@param  foregroundcolor  foreground colour to use
149             *@param  backgroundcolor  background colour to use
150             */
151            public SpeechBubble(int width, int height, int direction, String text, Color foregroundcolor, Color backgroundcolor) {
152                    super();
153    
154                    bg = backgroundcolor;
155                    fg = foregroundcolor;
156    
157                    JScrollBar scrollbar;
158    
159                    //Create rendering area
160                    textpanel = new JPanel();
161                    textpanel.setBackground(backgroundcolor);
162                    textpanel.setForeground(backgroundcolor);
163                    setBackground(backgroundcolor);
164                    setForeground(foregroundcolor);
165                    rh = new RenderingHints(RenderingHints.KEY_ANTIALIASING,
166                                    RenderingHints.VALUE_ANTIALIAS_ON);
167                    tk = getToolkit();
168    
169                    //If speech bubble peak is on the left
170                    if (direction == LEFT) {
171    
172                            //Construct text area
173                            textpanel.setLayout(new BoxLayout(textpanel, BoxLayout.Y_AXIS));
174                            this.setLayout(new BoxLayout(this, BoxLayout.X_AXIS));
175                            this.add(Box.createRigidArea(new Dimension((int) (width / 7f), 0)));
176                            textpanel.add(Box.createRigidArea(new Dimension(0, (int) (height / 25f))));
177                            sb = new RoundRectangle2D.Float(0f + 0.1f, 0f + 0.02f, (1f - 0.0025f) - 0.1f, (0.95f), 0.1f, 0.1f);
178                            this.setPreferredSize(new Dimension(width, height));
179    
180                            //Construct speech bubble outline
181                            speechtri = new GeneralPath();
182                            speechtri.moveTo(0.1f, 0.77f);
183                            speechtri.lineTo(0f + 0.02f, 0.88f);
184                            speechtri.lineTo(0.1f, 0.86f);
185                            erase = new Rectangle2D.Float(0.1f, 0.77f, 0.5f, 0.09f);
186                            marginleft = ((float) width / 10f);
187                            insetleft = ((float) width / 20f);
188                            insetright = insetleft;
189                            insettop = ((float) height / 40f);
190                            insetbottom = insettop;
191                            formatWidth = (float) width - (marginleft + insetleft + insetright);
192                            formatHeight = ((float) height) - (insettop + insetbottom);
193                            textpane = new JTextPane();
194                            textpane.setEditable(false);
195                            textpane.setForeground(foregroundcolor);
196                            textpane.setBackground(backgroundcolor);
197    
198                            //Construct scroll pane
199                            scrollpane = new JScrollPane(textpane);
200                            scrollpane.setVerticalScrollBarPolicy(JScrollPane.VERTICAL_SCROLLBAR_ALWAYS);
201                            scrollpane.setHorizontalScrollBarPolicy(JScrollPane.HORIZONTAL_SCROLLBAR_NEVER);
202                            scrollpane.setBackground(backgroundcolor);
203                            scrollpane.setForeground(backgroundcolor);
204                            scrollbar = scrollpane.getVerticalScrollBar();
205                            scrollbar.setBorder(BorderFactory.createLineBorder(Color.darkGray));
206                            scrollpane.setVerticalScrollBar(scrollbar);
207                            scrollpane.setViewportBorder(BorderFactory.createEmptyBorder());
208                            scrollpane.setBorder(BorderFactory.createEmptyBorder());
209    
210                            //Put it all together
211                            textpanel.add(scrollpane);
212                            textpanel.add(Box.createRigidArea(new Dimension(0, (int) (2f * insetbottom))));
213                            this.add(textpanel);
214                            this.add(Box.createRigidArea(new Dimension((int) (width / 35f), 0)));
215                    } else if (direction == BOTTOM) {
216    
217                            //If speech bubble peak is on the bottom
218                            textpanel.setLayout(new BoxLayout(textpanel, BoxLayout.Y_AXIS));
219                            this.setLayout(new BoxLayout(this, BoxLayout.X_AXIS));
220    
221                            //Construct speech bubble outline
222                            sb = new RoundRectangle2D.Float(0f + 0.04f, 0f + 0.04f, 1f - 0.08f, 0.83f, 0.1f, 0.1f);
223                            this.setPreferredSize(new Dimension(width, height));
224                            speechtri = new GeneralPath();
225                            speechtri.moveTo(0.72f, 0.87f);
226                            speechtri.lineTo(0.7f, 0.98f);
227                            speechtri.lineTo(0.79f, 0.87f);
228                            erase = new Rectangle2D.Float(0.72f, 0.86f, 0.07f, 0.01f);
229                            insetleft = ((float) width / 18f);
230                            insetright = ((float) width / 16f);
231                            insettop = ((float) height / 15f);
232                            insetbottom = ((float) height / 6.5f);
233    
234                            //Construct text area
235                            textpane = new JTextPane();
236                            textpane.setEditable(false);
237                            textpane.setForeground(foregroundcolor);
238                            textpane.setBackground(backgroundcolor);
239    
240                            //Construct scroll pane
241                            scrollpane = new JScrollPane(textpane);
242                            scrollpane.setVerticalScrollBarPolicy(JScrollPane.VERTICAL_SCROLLBAR_ALWAYS);
243                            scrollpane.setHorizontalScrollBarPolicy(JScrollPane.HORIZONTAL_SCROLLBAR_NEVER);
244                            scrollpane.setBackground(backgroundcolor);
245                            scrollpane.setForeground(backgroundcolor);
246                            scrollbar = scrollpane.getVerticalScrollBar();
247                            scrollbar.setBorder(BorderFactory.createLineBorder(backgroundcolor));
248                            scrollpane.setVerticalScrollBar(scrollbar);
249                            scrollpane.setViewportBorder(BorderFactory.createEmptyBorder());
250                            scrollpane.setBorder(BorderFactory.createEmptyBorder());
251    
252                            //Put it all together
253                            textpanel.add(Box.createRigidArea(new Dimension(0, (int) (insettop))));
254                            textpanel.add(scrollpane);
255                            textpanel.add(Box.createRigidArea(new Dimension(0, (int) (insetbottom))));
256                            this.add(Box.createRigidArea(new Dimension((int) (insetleft), 0)));
257                            this.add(textpanel);
258                            this.add(Box.createRigidArea(new Dimension((int) (insetright), 0)));
259                    }
260                    setText(text);
261                    super.repaint();
262            }

385            /**
386             *  Sets the text of the SpeechBubble object to be displayed
387             *
388             *@param  text  The new speech text
389             */
390            public void setText(String text) {
391                    plaintext = new String(text);
392                    if (plaintext.length() > 0) {
393                            textpane.setText(plaintext);
394                            this.repaint();
395                    }
396            }

399            /**
400             *  Appends to the text of the SpeechBubble object
401             *
402             *@param  text  The speech text to append
403             */
404            public void appendText(String text) {
405                    plaintext = plaintext.concat(text);
406                    if (plaintext.length() > 0) {
407                            textpane.setText(plaintext);
408                            this.repaint();
409                    }
410            }

413            /**
414             *  Paints the speech bubble component
415             *
416             *@param  graphics  The Graphics2d object for this panel
417             */
418            public void paintComponent(java.awt.Graphics graphics) {
419                    super.paintComponent(graphics);
420                    paintOverComponent(graphics);
421            }
422    
423    
424            /**
425             *  Paints over what has already been drawn in this panel. This method does not
426             *  "wipe" the speech bubble panel before drawing.
427             *
428             *@param  graphics  The Graphics2d object for this panel
429             */
430            public void paintOverComponent(java.awt.Graphics graphics) {
431                    int width;
432                    int height;
433                    boolean fontfit;
434    
435                    //Set rendering variables
436                    g2d = (Graphics2D) graphics;
437                    g2d.setRenderingHints(rh);
438                    width = getWidth();
439                    height = getHeight();
440                    g2d.setStroke(new BasicStroke(0.01f, BasicStroke.CAP_SQUARE, BasicStroke.JOIN_ROUND));
441    
442                    //Paint the speech bubble
443                    g2d.scale(width, height);
444                    g2d.setPaint(bg);
445                    g2d.fill(sb);
446                    g2d.setPaint(fg);
447                    g2d.draw(sb);
448                    g2d.setPaint(bg);
449                    g2d.fill(speechtri);
450                    g2d.draw(speechtri);
451                    g2d.fill(erase);
452                    g2d.draw(erase);
453                    g2d.setPaint(fg);
454                    g2d.draw(speechtri);
455    
456                    //Draw text
457                    scrollpane.scrollRectToVisible(new Rectangle(0, 0, 0, 0));
458                    scrollpane.repaint();
459                    tk.sync();
460            }
