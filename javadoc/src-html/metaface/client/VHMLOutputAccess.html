001    package metaface.client;
002    import javax.xml.parsers.*;
003    import org.xml.sax.*;
004    import org.xml.sax.helpers.*;
005    import java.util.*;
006    import java.io.*;
007    import java.security.*;
008    
009    /**
010     *  This class allows XML parsing errors to propogate due to privileged actions.
011     *  Otherwise a security exception will be thrown for XML errors.
012     *
013     *@author     Simon Beard 
014     *@version    1.0
015     *@see        VHMLOutputTagReader
016     *@see        VHMLOutputReader
017     */
018    class VHMLOutputAccess implements PrivilegedAction {
019            private XMLReader xmlreader;
020            private String eca_synthesis_file;
021    
022    
023            /**
024             *  Constructor for the VHMLOutputAccess object
025             *
026             *@param  xmlr  The XML parser needing privileged action
027             *@param  file  The VHML output file to be parsed
028             */
029            public VHMLOutputAccess(XMLReader xmlr, String file) {
030                    xmlreader = xmlr;
031                    eca_synthesis_file = file;
032            }
033    
034    
035            /**
036             *  Main processing method for the VHMLOutputAccess object in order to execute
037             *  a privileged action
038             *
039             *@return    null
040             */
041            public Object run() {
042                    try {
043                            xmlreader.parse(eca_synthesis_file);
044                    } catch (SAXException e) {
045                            System.err.println("XML error parsing VHML output file");
046                            e.printStackTrace();
047                    } catch (IOException e) {
048                            System.err.println("IO error parsing VHML output file");
049                            e.printStackTrace();
050                    }
051                    return null;
052            }
053    }
