001    package metaface.client;
002    
003    import java.awt.*;
004    import java.applet.*;
005    import java.awt.event.*;
006    import javax.swing.*;
007    import netscape.javascript.*;
008    import java.net.*;
009    import java.io.*;
010    
011    import metaface.embodiment.*;
012    
013    /**
014     *  This default class handles the instantiation of a MetaFace applet within the
015     *  browser
016     *
017     *@author     Simon Beard 
018     *@version    1.0
019     */
020    public abstract class DefaultMetaFaceApplet extends MetaFaceApplet {
021    
022            /**
023             *  The speech bubble for the talking head
024             */
025            private SpeechBubble speechBubble;
026            /**
027             *  The dialogue input object
028             */
029            private TextControl tc;
030            /**
031             *  The MPEG-4 face drawing region
032             */
033            private DrawingArea appletDA;
034            /**
035             *  The MetaFace client to use
036             */
037            private MetaFaceClient dmfc = null;
038    
039    
040            /**
041             *  Initializes the applet MetaFaceApplet
042             */
043            public DefaultMetaFaceApplet() {
044                    super.init();
045            }
046    
047    
048            /**
049             *  Sets the client attribute of the DefaultMetaFaceApplet object
050             *
051             *@param  mfc  The new client
052             */
053            public void setClient(MetaFaceClient mfc) {
054                    dmfc = mfc;
055            }
056    
057    
058            /**
059             *  Initialise the MetaFace applet
060             */
061            public void init() {
062                    JPanel one;
063                    JPanel two;
064                    JPanel three;
065                    JPanel four;
066                    Rectangle bounds;
067                    int height;
068    
069                    super.init();
070    
071                    // Create GUI and setup
072                    setBackground(java.awt.Color.white);
073                    bounds = this.getBounds();
074    
075                    // Create dialogue input object
076                    tc = new TextControl((int) (getBounds().getWidth()));
077                    tc.setBackground(java.awt.Color.black);
078                    tc.setForeground(java.awt.Color.black);
079                    tc.setBackground(new Color(255, 255, 255));
080                    height = tc.getHeight();
081                    appletDA = new DrawingArea((int) (bounds.getWidth()), (int) ((bounds.getHeight() / 2f) - height));
082                    height = 30;
083    
084                    //Speech bubble for displaying speech text
085                    speechBubble = new SpeechBubble((int) (bounds.getWidth()), (int) ((bounds.getHeight() / 2f)), SpeechBubble.BOTTOM, "Welcome to Simon Beard's Resume");
086    
087                    //Set layou of components
088                    getContentPane().setBackground(java.awt.Color.black);
089                    getContentPane().setLayout(new BorderLayout(0, 0));
090                    one = new JPanel();
091                    two = new JPanel();
092                    three = new JPanel();
093                    four = new JPanel();
094                    one.setBackground(Color.black);
095                    two.setBackground(Color.black);
096                    three.setBackground(Color.black);
097                    four.setBackground(Color.black);
098                    three.setLayout(new BoxLayout(three, BoxLayout.Y_AXIS));
099                    four.setLayout(new FlowLayout(FlowLayout.CENTER));
100                    two.setLayout(new BoxLayout(two, BoxLayout.Y_AXIS));
101                    two.add(appletDA);
102                    two.add(four);
103                    three.add(speechBubble);
104                    three.add(two);
105                    getContentPane().add(three, BorderLayout.CENTER);
106            }
107    
108    
109            /**
110             *  Start the applet
111             */
112            public void start() {
113                    try {
114                            dmfc = new DefaultMetaFaceClient(appletDA, JSObject.getWindow(this));
115                    } catch (Exception e) {
116                            e.printStackTrace();
117                    }
118            }
119    
120    
121            /**
122             *  Initialise some components
123             */
124            private void initComponents() {
125                    setLayout(new java.awt.BorderLayout());
126                    setBackground(java.awt.Color.black);
127            }
128    
129    
130            /**
131             *  This method is called by web browser JavaScript methods to notify of URL
132             *  changes to target frames
133             *
134             *@param  url     The new URL displayed
135             *@param  target  The target frame of the web browser
136             */
137            public void updateWebPage(String url, String target) {
138    
139                    //Notify client of changes
140                    if ((url != null) && (target != null)) {
141                            if (dmfc != null) {
142                                    dmfc.updateWebPage(url, target);
143                            }
144                    }
145            }
146    }
